```php
class CtfController extends Controller
{
    public function __construct(
        protected CtfService $ctfService,
        protected CtfScoreboardService $scoreboardService,
    ) {}

    public function index(): View
    {
        $competitions = CtfCompetition::published()
            ->public()
            ->orderByDesc('start_date')
            ->withCount('challenges')
            ->get()
            ->map(function (CtfCompetition $c) {
                $c->solvedCount = $c->challenges()
                    ->whereHas('submissions', fn ($q) => $q->where('user_id', auth()->id())->where('is_correct', true))
                    ->count();
                return $c;
            });

        return view('pages.ctf.index', [
            'title' => 'CTF Arena',
            'competitions' => $competitions,
        ]);
    }

    public function show(CtfCompetition $competition): View
    {
        abort_unless($competition->status === 'published' || auth()->user()->isAdmin(), 403);

        $challenges = $competition->challenges()
            ->where('is_active', true)
            ->with('category')
            ->orderBy('sort_order')
            ->orderBy('points')
            ->get()
            ->groupBy(fn ($c) => $c->category->name);

        $userSolved = CtfSubmission::where('user_id', auth()->id())
            ->whereIn('ctf_challenge_id', $competition->challenges()->pluck('id'))
            ->where('is_correct', true)
            ->pluck('ctf_challenge_id')
            ->toArray();

        $scoreboard = $this->scoreboardService->getScoreboard($competition, 10);

        return view('pages.ctf.competition', [
            'title' => $competition->title,
            'competition' => $competition,
            'challengesByCategory' => $challenges,
            'userSolved' => $userSolved,
            'scoreboard' => $scoreboard,
        ]);
    }

    public function submit(
        SubmitCtfFlagRequest $request,
        CtfCompetition $competition,
        CtfChallenge $challenge
    ): RedirectResponse {
        $result = $this->ctfService->submitFlag(
            $challenge,
            auth()->user(),
            $request->string('flag'),
            $request->ip()
        );

        if ($result['success']) {
            return redirect()
                ->back()
                ->with('status', "Correct! +{$result['points']} points. Total: {$result['total_points']}");
        }

        return redirect()
            ->back()
            ->with('error', $result['message']);
    }

    public function scoreboard(CtfCompetition $competition): View
    {
        $scoreboard = $this->scoreboardService->getScoreboard($competition, 100);
        $userRank = $this->scoreboardService->getUserRank($competition, auth()->user());

        return view('pages.ctf.scoreboard', [
            'title' => "{$competition->title} — Scoreboard",
            'competition' => $competition,
            'scoreboard' => $scoreboard,
            'userRank' => $userRank,
        ]);
    }

    public function writeup(CtfCompetition $competition, CtfChallenge $challenge): View
    {
        abort_unless($challenge->isSolvedBy(auth()->user()), 403);

        $existingWriteup = CtfWriteup::where('ctf_challenge_id', $challenge->id)
            ->where('user_id', auth()->id())
            ->first();

        return view('pages.ctf.writeup', [
            'title' => "Writeup: {$challenge->title}",
            'competition' => $competition,
            'challenge' => $challenge,
            'existingWriteup' => $existingWriteup,
        ]);
    }

    public function submitWriteup(
        StoreCtfWriteupRequest $request,
        CtfCompetition $competition,
        CtfChallenge $challenge
    ): RedirectResponse {
        abort_unless($challenge->isSolvedBy(auth()->user()), 403);

        CtfWriteup::updateOrCreate(
            ['ctf_challenge_id' => $challenge->id, 'user_id' => auth()->id()],
            ['content' => $request->string('content'), 'status' => 'pending']
        );

        return redirect()
            ->route('ctf.competition', $competition)
            ->with('status', 'Writeup submitted for review.');
    }
}
```

```php
class Scoreboard extends Component
{
    public CtfCompetition $competition;
    public Collection $scoreboard;
    public ?int $userRank;

    public function mount(CtfCompetition $competition): void
    {
        $this->competition = $competition;
        $this->loadScoreboard();
    }

    public function loadScoreboard(): void
    {
        $service = app(CtfScoreboardService::class);
        $this->scoreboard = $service->getScoreboard($this->competition, 100);
        $this->userRank = $service->getUserRank($this->competition, auth()->user());
    }

    public function render(): View
    {
        return view('livewire.ctf.scoreboard', [
            'scoreboard' => $this->scoreboard,
            'userRank' => $this->userRank,
        ]);
    }
}
```

```php
class ChallengeDetails extends Component
{
    public CtfCompetition $competition;
    public CtfChallenge $challenge;
    public string $flag = '';
    public ?string $result = null;
    public ?string $resultType = null; // 'success' or 'error'

    public function mount(CtfCompetition $competition, CtfChallenge $challenge): void
    {
        $this->competition = $competition;
        $this->challenge = $challenge;
    }

    public function submitFlag(): void
    {
        $this->validate(['flag' => 'required|string|min:3|max:255']);

        $service = app(CtfService::class);
        $result = $service->submitFlag($this->challenge, auth()->user(), $this->flag, request()->ip());

        $this->result = $result['success']
            ? "Correct! +{$result['points']} points"
            : $result['message'];
        $this->resultType = $result['success'] ? 'success' : 'error';

        if ($result['success']) {
            $this->flag = '';
            $this->dispatch('ctf-challenge-solved')->to(Scoreboard::class);
        }
    }

    public function render(): View
    {
        $isSolved = $this->challenge->isSolvedBy(auth()->user());

        return view('livewire.ctf.challenge-details', [
            'challenge' => $this->challenge,
            'isSolved' => $isSolved,
            'category' => $this->challenge->category,
        ]);
    }
}
```

```php
class WriteupForm extends Component
{
    public CtfCompetition $competition;
    public CtfChallenge $challenge;
    public ?CtfWriteup $existingWriteup;
    public string $content = '';
    public bool $submitted = false;

    public function mount(CtfCompetition $competition, CtfChallenge $challenge): void
    {
        $this->competition = $competition;
        $this->challenge = $challenge;
        $this->existingWriteup = CtfWriteup::where('ctf_challenge_id', $challenge->id)
            ->where('user_id', auth()->id())
            ->first();
        if ($this->existingWriteup) {
            $this->content = $this->existingWriteup->content;
            $this->submitted = true;
        }
    }

    public function submit(): void
    {
        $this->validate(['content' => 'required|string|min:100|max:50000']);

        CtfWriteup::updateOrCreate(
            ['ctf_challenge_id' => $this->challenge->id, 'user_id' => auth()->id()],
            ['content' => $this->content, 'status' => 'pending']
        );

        $this->submitted = true;
        session()->flash('status', 'Writeup submitted for review.');
    }

    public function render(): View
    {
        return view('livewire.ctf.writeup-form');
    }
}
```
